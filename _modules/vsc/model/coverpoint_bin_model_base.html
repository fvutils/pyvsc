    def finalize(self, bin_idx_base:int)->int:
        """Accepts the bin index where this bin starts ; returns number of bins"""
        cp = self.parent
        while cp is not None and not isinstance(cp, CoverpointModel):
            cp = cp.parent
        self.cp = cp
        
        self.bin_idx_base = bin_idx_base
        
        return 0
