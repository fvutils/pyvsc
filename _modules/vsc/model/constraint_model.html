    @staticmethod
    def and_nodelist(node_l, btor):
        """Creates a boolean AND across a list of expression nodes"""
        ret = None
        ret_valid = False
        
        for n in node_l:
            if ret_valid:
                ret = btor.And(ret, n)
            else:
                ret = n
                ret_valid = True
        
        return ret
